      <input type="text" id="staffRole" placeholder="Role/Position">
      <button onclick="addStaff()">Add Staff</button>
      <ul id="staffList"></ul>
    </section>

    <section>
      <h2>Task Tracker</h2>
      <input type="text" id="taskTitle" placeholder="Task Title">
      <textarea id="taskDetails" placeholder="Task Details"></textarea>
      <button onclick="addTask()">Add Task</button>
      <ul id="taskList"></ul>
    </section>

    <section>
      <h2>Admin Notes</h2>
      <textarea id="noteText" placeholder="Write note here..."></textarea>
      <button onclick="saveNote()">Save Note</button>
      <div id="savedNotes"></div>
    </section>

  <script>
    function addStaff() {
      const name = document.getElementById('staffName').value;
      const role = document.getElementById('staffRole').value;
      if (name && role) {
        const li = document.createElement('li');
        li.textContent = `${name} â€” ${role}`;
        document.getElementById('staffList').appendChild(li);
        document.getElementById('staffName').value = '';
        document.getElementById('staffRole').value = '';
        saveData();
      }
    }

    function addTask() {
      const title = document.getElementById('taskTitle').value;
      const details = document.getElementById('taskDetails').value;
      if (title) {
        const li = document.createElement('li');
        li.textContent = `${title}: ${details}`;
        document.getElementById('taskList').appendChild(li);
        document.getElementById('taskTitle').value = '';
        document.getElementById('taskDetails').value = '';
        saveData();
      }
    }

    function saveNote() {
      const text = document.getElementById('noteText').value;
      if (text) {
        const p = document.createElement('p');
        p.textContent = text;
        document.getElementById('savedNotes').appendChild(p);
        document.getElementById('noteText').value = '';
        saveData();
      }
    }

    function saveData() {
      const data = {
        staff: document.getElementById('staffList').innerHTML,
        tasks: document.getElementById('taskList').innerHTML,
        notes: document.getElementById('savedNotes').innerHTML
      };
      localStorage.setItem('unitAdminData', JSON.stringify(data));
    }

    function loadData() {
      const saved = localStorage.getItem('unitAdminData');
      if (saved) {
        const data = JSON.parse(saved);
        document.getElementById('staffList').innerHTML = data.staff;
        document.getElementById('taskList').innerHTML = data.tasks;
        document.getElementById('savedNotes').innerHTML = data.notes;
      }
    }

    window.onload = loadData;
  </script>
